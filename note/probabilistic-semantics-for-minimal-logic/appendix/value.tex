p,q\in{}U & \Longrightarrow & \left|p\rightarrow{}q\right| & := & \left|q\right|_{}^{\left|p\right|}                                                        \\

          &                 & \left|\makebox{\bot}\right|  & := & 0                                                                                         \\
\end{array}
